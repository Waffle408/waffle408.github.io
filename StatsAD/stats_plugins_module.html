How STATS generates overdue reports

A scheduled job (or on-demand action) queries open assignments that exceed their allowed borrow window. The plugin then renders a CSV/HTML table and can queue notification emails to year advisors or asset owners. You can tune grace periods, recipients, and filters per year level. 

// PSEUDOCODE (frontend-friendly) — no PHP
async function generateOverdueReport() {
  // 1) Fetch open assignments with due dates in the past
  const res = await fetch('/api/overdue?daysGrace=1'); // e.g. 1-day grace
  const items = await res.json(); // [{ student_id, student_name, device_id, due, days_over }]

  // 2) Render a table or export CSV
  renderTable('#overdueTable', items);

  // 3) Optionally queue emails (batched to avoid rate limits)
  const emails = items.map(it => ({ to: advisorFor(it.student_id), subject: 'Overdue device', body: buildBody(it) }));
  if (emails.length) await fetch('/api/report_emails/queue', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ emails })
  });

  // 4) Persist report metadata for audit / downloads
  await fetch('/api/reports', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      filename: exportCsv(items),
      generated_by: currentUserId(),
      filters_used: JSON.stringify({ type: 'overdue', daysGrace: 1 })
    })
  });
}
